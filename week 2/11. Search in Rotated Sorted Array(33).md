class Solution {
    public int search(int[] nums, int target) {
        int index = indexOf(nums, target);
        return index;
        
    }
     public static int indexOf(int[] array, int target) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == target) {
                return i;
            }
        }
        return -1; 
    }
}
